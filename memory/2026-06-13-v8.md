## Technical Implementation (v8)

### New JavaScript Functions

```javascript
// Detect driver location
function getDriverLocation() {
  if (navigator.geolocation) {
    navigator.geolocation.getCurrentPosition(
      (position) => {
        driverLocation = {
          lat: position.coords.latitude,
          lng: position.coords.longitude
        };
        sortAndRender();
      },
      (error) => {
        driverLocation = { lat: 43.6629, lng: -79.3957 }; // fallback
        sortAndRender();
      }
    );
  }
}

// Sort by type + proximity
function sortAndRender() {
  // Calculate distances
  currentStops.forEach(stop => {
    stop.distance = haversine(driverLocation.lat, driverLocation.lng, 
                              stop.lat, stop.lng);
  });
  
  // Sort: rollin first, then by distance
  currentStops.sort((a, b) => {
    if (a.type !== b.type) {
      return a.type === 'rollin' ? -1 : 1;
    }
    return a.distance - b.distance;
  });
  
  // Assign sequence numbers
  sequenceNumbers = {};
  currentStops.forEach((stop, idx) => {
    sequenceNumbers[stop.id] = idx + 1;
  });
  
  renderServiceList();
  renderMarkers(currentStops);
  updateMapCenter();
}

// Show driver location on map
function updateMapCenter() {
  map.setCenter(driverLocation);
  new google.maps.Marker({
    position: driverLocation,
    icon: { scale: 12, fillColor: '#3b82f6' }
  });
}
```
